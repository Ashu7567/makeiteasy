  function setStatus(text) {

    status.textContent = text;

    clearTimeout(window.__statusTimer);

    window.__statusTimer = setTimeout(() => status.textContent = 'Ready', 1600);

  }


  function parseDetails(text) {

    const result = {};

    const parsed = [];

    const unknown = [];

    const malformed = [];


    text.split(/\r?\n/).forEach((line, idx) => {

      const trimmed = line.trim();

      if (!trimmed) return;

      const sep = trimmed.indexOf(',');

      if (sep === -1) {

        malformed.push({ line: idx + 1, text: trimmed });

        return;

      }

      const label = trimmed.slice(0, sep).trim();

      const value = trimmed.slice(sep + 1).trim();

      const key = labelToKey[label];

      parsed.push({ label, key: key || null, value });

      if (!key) {

        unknown.push({ line: idx + 1, label, value });

        return;

      }

      result[key] = value;

    });


    return { result, parsed, unknown, malformed };

  }


  function fillMissingRules(data) {

    const out = { ...data };


    out.integrationName = ""; // always blank


    // company = tenant

    if (keepCompanySameAsTenant.checked) {

      out.companyName = out.tenantName || out.companyName || "";

    }


    // NEW RULE: apiGatewayURL = vanityURL

    if (out.vanityURL) {

      out.apiGateWayURL = out.vanityURL;

    }


    if (!out.sourceTenant) out.sourceTenant = defaults.sourceTenant;

    if (!out.serviceAccountUser) out.serviceAccountUser = defaults.serviceAccountUser;

    if (!out.sftpVanityURL) out.sftpVanityURL = defaults.sftpVanityURL;


    return out;

  }


  function renderWarnings(info, templateKeys, merged) {

    warningsEl.innerHTML = '';

    statsEl.innerHTML = '';


    const missing = templateKeys.filter(k => {

      const val = merged[k];

      return val === undefined || val === null || val === '';

    });


    const items = [];


    if (info.malformed.length) {

      info.malformed.forEach(item => {

        items.push(`<li class="bad">Line ${item.line}: invalid format (use Label,Value) → ${escapeHtml(item.text)}</li>`);

      });

    }

    if (info.unknown.length) {

      info.unknown.forEach(item => {

        items.push(`<li class="warn">Line ${item.line}: unknown label ignored → ${escapeHtml(item.label)}</li>`);

      });

    }

    if (missing.length) {

      missing.forEach(key => {

        items.push(`<li class="warn">Missing value for template field → ${escapeHtml(key)}</li>`);

      });

    }

    if (!items.length) {

      items.push('<li class="good">No warnings. Mapping looks good.</li>');

    }


    warningsEl.innerHTML = items.join('');

    statsEl.innerHTML = `

      <span class="pill good">Mapped: ${Object.keys(info.result).length}</span>

      <span class="pill warn">Unknown labels: ${info.unknown.length}</span>

      <span class="pill bad">Malformed lines: ${info.malformed.length}</span>

      <span class="pill">Missing fields: ${missing.length}</span>

    `;

  }


  function escapeHtml(str) {

    return String(str)

      .replace(/&/g, '&amp;')

      .replace(/</g, '&lt;')

      .replace(/>/g, '&gt;')

      .replace(/"/g, '&quot;')

      .replace(/'/g, '&#39;');

  }


  function fillJSON() {

    try {

      const template = JSON.parse(jsonInput.value);

      const templateKeys = Object.keys(template);

      const parsed = parseDetails(detailsInput.value);

      const merged = fillMissingRules(parsed.result);


      // Only fill keys that already exist in the template.

      templateKeys.forEach(key => {

        if (Object.prototype.hasOwnProperty.call(merged, key)) {

          template[key] = merged[key];

        }

      });


      output.value = JSON.stringify(template, null, 2);

      renderWarnings(parsed, templateKeys, merged);

      setStatus('Filled');

    } catch (e) {

      warningsEl.innerHTML = `<li class="bad">Invalid JSON template. Please check braces/quotes.</li>`;

      statsEl.innerHTML = '';

      alert('Invalid JSON template ya details format.');

    }

  }


  async function copyOutput() {

    try {

      await navigator.clipboard.writeText(output.value || '');

      setStatus('Copied');

    } catch (e) {

      output.select();

      document.execCommand('copy');

      setStatus('Copied');

    }

  }
